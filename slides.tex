\begin{frame}{Frá námi til starfs – samhengi}
\begin{itemize}
  \item Háskólar vinna markvisst að því að tengja nám við starfsumhverfi.
  \item Atvinnulífið krefst:
  \begin{itemize}
    \item gagnsærrar samvinnu,
    \item rekjanleika og endurtakanleika,
    \item fagmennsku í framsetningu og aðferðavali.
  \end{itemize}
  \item Verkefnadrifin námskeið bjóða upp á mikla möguleika — ef ferlarnir eru sýnilegir.
\end{itemize}
\end{frame}

% ----------------------------------------------------
\begin{frame}{Hvað lærum við af atvinnulífinu?}
\begin{itemize}
  \item Vinnulag sem sameinar:
  \begin{itemize}
    \item \emph{agile} vinnubrögð,
    \item stuttir sprettir,
    \item samvinnudrifna ákvarðanatöku.
  \end{itemize}
  \item Tól sem styðja þetta:
  \begin{itemize}
    \item Git, GitHub, Issues, Branches, Pull Requests.
    \item GitHub Classroom til skipulags og verkefnastjórnunar.
  \end{itemize}
  \item Vandinn í hefðbundnum verkefnadrifnum námskeiðum:
  \begin{itemize}
    \item framlag einstaklings óljóst,
    \item framvinda ósýnileg,
    \item laumufarþegar og misvægi í teymum.
  \end{itemize}
\end{itemize}
\end{frame}

% ----------------------------------------------------
\begin{frame}{Pull requests (PR) sem matsinnviðir}
\begin{itemize}
  \item PR eru ekki bara tækni — heldur kennslugögn.
  \item PR binda saman:
  \begin{itemize}
    \item tæknilegar breytingar,
    \item rökstuðning,
    \item jafningjarýni,
    \item viðbrögð við endurgjöf.
  \end{itemize}
  \item Gerir framlag og ferla:
  \begin{itemize}
    \item sýnileg,
    \item rekjanleg,
    \item metanleg.
  \end{itemize}
  \item Lykilhugtök: \emph{efnisleg rýni}, \emph{viðbragðsskylda} (e. responsiveness), \emph{skipulögð verkafhending} (e. handover).
\end{itemize}
\end{frame}

% ----------------------------------------------------
\begin{frame}{Reglur sem móta námsmat}
\begin{enumerate}
  \item Skýr PR-lýsing og afmarkað umfang.
  \item Minna en tvær efnislegar rýnir ekki samþykkt.
  \item Allar athugasemdir svarað — engar ,,þöglar'' lagfæringar.
  \item ,,Merge'' aðeins við samþykki; ólokið fer í Issues.
  \item README og keyrsluleiðbeiningar uppfærðar við breytingar.
  \item Tímastimplar og framlag einstaklings skráist sjálfkrafa.
\end{enumerate}
\end{frame}

% ----------------------------------------------------
\begin{frame}{Reynsla úr tveimur námskeiðum}
\textbf{Upplýsingaverkfræði – 2. ár BS}
\begin{itemize}
  \item Stuttar lotur með áherslu á að læra vinnuferlana.
  \item Hver lota í sér geymslu og byrjar á skýrum upphafspunkti.
\end{itemize}

\textbf{Viðskiptagreind – 3. ár BS / 1. ár MS}
\begin{itemize}
  \item Ein viðvarandi geymsla allt misserið (samfella og rekjanleiki).
  \item Endurgjöf flyst sjálfkrafa milli lota (\emph{carry-forward}).
  \item Þemu námskeiðsins: fagmennska, rökstuðningur og traust í samskiptum við fyrirtæki.
\end{itemize}
\end{frame}

% ----------------------------------------------------
\begin{frame}{Fagmennska í Viðskiptagreind}
\begin{itemize}
  \item Samstarf við fyrirtæki með raunveruleg gögn.
  \item Sama gagnasafn allt misserið → dýpri greining.
  \item TBL/PBL + PR-miðað mat:
  \begin{itemize}
    \item iRAT, tRAT, tAPP, iREF,
    \item Capstone (tCAP + iCAP).
  \end{itemize}
  \item Nemendur læra að:
  \begin{itemize}
    \item skrá forsendur,
    \item bregðast við athugasemdum,
    \item sýna trúverðugleika í kynningum.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Ávinningur fyrir nemendur}
\begin{itemize}
  \item Betri yfirsýn og samræming í teymum.
  \item Sýnilegt framlag, dregur úr laumufarþegum.
  \item Skýrari væntingar í teymisvinnu.
  \item Vinnulag sem er beint yfirfært í atvinnulíf.
\end{itemize}
\end{frame}

% ----------------------------------------------------
\begin{frame}{Ávinningur fyrir kennara}
\begin{itemize}
  \item Auðveldara að sjá framvindu og einstaklingsframlög.
  \item Endurgjöf fer ekki forgörðum — hún er bundin við PR.
  \item Capstone endurmat verður skilvirkara (diff → ekki allt upp á nýtt).
  \item Betri leið til að meta fagmennsku.
\end{itemize}
\end{frame}

\end{itemize}
